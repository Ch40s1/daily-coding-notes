Sliding Window

          
            pseudocode:
            def slidingWindow(arr, k):

            // set left pointer to 0
              left = 0

            // set right pointer to 0
              right = 0

            // set the sum of the window to 0
              windowSum = 0

            // set the maximum sum to 0
              maxSum = 0

            // while the right pointer is less than the length of the array
              while right < len(arr):

            // add the element at the right pointer to the window sum
                windowSum += arr[right]

            // if the window size is greater than k
                if right >= k - 1:

            // set the maximum sum to the maximum of the window sum and the maximum sum
                  maxSum = max(maxSum, windowSum)

            // subtract the element at the left pointer from the window sum
                  windowSum -= arr[left]

            // move the left pointer to the right
                  left += 1

            // move the right pointer to the right
                right += 1

            // return the maximum sum
              return maxSum

Prefix Sum

          
            pseudocode:
            def prefixSum(arr):

            // create a prefix sum array
              prefixSum = [0] * len(arr)

            // set the first element of the prefix sum array to the first element of the input array
              prefixSum[0] = arr[0]

            // for each element in the input array
              for i in range(1, len(arr)):

            // set the element of the prefix sum array to the sum of the element and the previous element
                prefixSum[i] = prefixSum[i - 1] + arr[i]

            // return the prefix sum array
              return prefixSum
